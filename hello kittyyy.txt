    <script>
        // Sonsuz "Çok" Yazısı Oluşturma
        const textContainer = document.getElementById('scrolling-text');
        let text = "ÇOK ÇOK ÇOK ÇOK ÇOK ÇOK ÇOK ";
        for(let i=0; i<10; i++) { text += text; } // Metni devasa hale getir
        textContainer.innerText = text;

        let currentY = 0;
        function scrollText() {
            currentY--;
            if (Math.abs(currentY) >= textContainer.offsetHeight / 2) {
                currentY = 0;
            }
            textContainer.style.top = currentY + 'px';
            requestAnimationFrame(scrollText);
        }
        scrollText();

        // Hello Kitty Yağmuru
        const kitties = ["🎀🐱", "🤍🎀", "😻", "🌸", "🎀", "🍭"];
        function createKitty() {
            const kitty = document.createElement("div");
            kitty.className = "hello-kitty";
            kitty.innerText = kitties[Math.floor(Math.random() * kitties.length)];
            kitty.style.left = Math.random() * 100 + "vw";
            kitty.style.top = "-60px";
            const duration = Math.random() * 5 + 4;
            kitty.style.animation = `floatDown ${duration}s linear forwards`;
            document.body.appendChild(kitty);
            setTimeout(() => { kitty.remove(); }, duration * 1000);
        }
        setInterval(createKitty, 400);

        // Müzik ve Başlatma
        function playMusic() {
            const music = document.getElementById('bgMusic');
            const btn = document.querySelector('.music-btn');
            
            if (music.paused) {
                music.play();
                btn.innerText = "⏸ Müziği Durdur";
                btn.style.backgroundColor = "#ff1493";
            } else {
                music.pause();
                btn.innerText = "▶ Müziği Devam Et";
                btn.style.backgroundColor = "#ff69b4";
            }
        }
    </script>
